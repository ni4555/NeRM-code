import numpy as np

def heuristics_v2(prize: np.ndarray, weight: np.ndarray) -> np.ndarray:
    n, m = weight.shape
    normalized_weight = weight / weight.sum(axis=1, keepdims=True)
    utility = prize / normalized_weight.sum(axis=1, keepdims=True)
    max_utility = utility.max(axis=1)
    diversity = np.std(utility, axis=1)
    heuristic = max_utility + diversity
    # Adaptive normalization based on the sum of utilities
    normalization_factor = np.sum(heuristic)
    heuristic_normalized = heuristic / normalization_factor
    return heuristic_normalized
